def somme(P,Q):
    p,q=len(P),len(Q)
    #result est la liste des coeff du polynme P+Q
    result=[]
    if p>=q:
        result=P.copy()
        for k in range(q):
            result[k]=result[k]+Q[k]
    else:
        result=Q.copy()
        for k in range(p):
            result[k]=result[k]+P[k]
    #retrait des ventuels 0 en fin de liste, en prenant la convention que la liste du polynme nul est [0]
    while result[-1]==0 and len(result)>1:
        result.pop(-1)
    return result

def euclide(P,Q):
    q=degre(Q)
    while degre(P)>=degre(Q):
        p=degre(P)
        #cration de la liste associe au polynme B=-X^{p-q}dom(P)/dom(Q)
        B=[0]*(p-q)
        B.append(-P[-1]/Q[-1])
        #P+BQ pour retirer le monme dominant de P
        P=somme(P,produit(B,Q))
    return P

Q1) Commenons par dterminer une quation cartsienne de la droite $(AB)$ :
Soit $M(x,y)$ un point quelconque du plan. Alors on a les quivalences suivantes :
$M(x,y)\in (AB) \Leftrightarrow \det(\overrightarrow{AM},\overrightarrow{AB})=0_{\mathbb{R}} \Leftrightarrow \begin{vmatrix} x-a & b-a \\ y-f(a) & f(b)-f(a) \end{vmatrix}=0_{\mathbb{R}}\Leftrightarrow (x-a)(f(b)-f(a))-(y-f(a))(b-a)=0_{\mathbb{R}}$
Ainsi, une quation cartsienne de la droite $(AB)$ est donne par : $(f(b)-f(a))x+(a-b)y+(-a(f(b)-f(a))+f(a)(b-a))=0_{\mathbb{R}}$.
En nullifiant $y$, il vient que l'abscisse c du point d'intersection de la droite $(AB)$ avec l'axe des abscisses est donn par :

$c=a-\dfrac{f(a)(b-a)}{f(b)-f(a)}=\dfrac{af(b)-bf(a)}{f(b)-f(a)}$

def appart(L,M):
    p,n = len(L),len(M)
    for i in range(n-p+1):
        j=0
        while j<p and L[j]==M[i+j]:
            j=j+1
        if j==p:
            return i
    return -1

def coupure(L):
    M=[]
    i=0
    while i<len(L):
        start=L[i]
        j=1
        while i+j<len(L) and L[i+j]==start:
            j=j+1
        if i+j<len(L):
            M.append(i+j)
        i=i+j
    return M

La liste donnant la dcomposition en facteurs irrductibles de 360 est [[2, 3], [3, 2], [5, 1]]
Le nombre de diviseurs de 360 est 24

Soit $n=\prod_{i}p_i^{\alpha_i}$ la dcomposition en produit de facteurs irrductibles de l'entier $n$. Alors le nombre de diviseurs de $n$ est donn par $\prod_{i}(\alpha_i+1)$.
Par exemple, $360=2^3\times 3^2\times 5^{1}$ admet $(3+1)\times(2+1)\times(1+1)=24$ diviseurs.
